function livrosPorCategoria($livros, $categoria)
{
    if ($categoria === 'Todos') {
        return $livros;
    }
    return array_filter($livros, function ($livro) use ($categoria) {
        return strtolower($livro['categoria']) === strtolower($categoria);
    });
}
